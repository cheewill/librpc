\textcolor{preprocessor}{#include <stdio.h>}

\textcolor{keywordtype}{int}
main(\textcolor{keywordtype}{int} argc, \textcolor{keyword}{const} \textcolor{keywordtype}{char} *argv[])
\{
    \textcolor{keyword}{struct }\hyperlink{structrpc__bus__node}{rpc\_bus\_node} *nodes;
    \textcolor{keywordtype}{int} n, i;
    \textcolor{keywordtype}{int} ret;

    (void)argc;
    (void)argv;

    \hyperlink{bus_8h_a845c8c255ac97bb03ac9ad1418eb66f0}{rpc\_bus\_open}();

    n = \hyperlink{bus_8h_a7dfb66a85e5f3037635a95b693d39a97}{rpc\_bus\_enumerate}(&nodes);
    \textcolor{keywordflow}{if} (n < 0) \{
        fprintf(stderr, \textcolor{stringliteral}{"rpc\_bus\_enumerate: %s\(\backslash\)n"}, strerror(errno));
        \textcolor{keywordflow}{return} (1);
    \}

    \textcolor{keywordflow}{for} (i = 0; i < n; i++) \{
        printf(\textcolor{stringliteral}{"%d: %s (%s)\(\backslash\)n"}, nodes[i].\hyperlink{structrpc__bus__node_a624e2a0be665938924ab2f815afb669e}{rbn\_address},
            nodes[i].\hyperlink{structrpc__bus__node_a9afbf9a155ae01c1c47dc210a8334022}{rbn\_name}, nodes[i].\hyperlink{structrpc__bus__node_af5d23beee7c50520e7235f0c922fafb1}{rbn\_description});

        ret = \hyperlink{bus_8h_ac4f30e57bd3b17725db2e517e7cf72a4}{rpc\_bus\_ping}(nodes[i].rbn\_name);
        \textcolor{keywordflow}{switch} (ret) \{
        \textcolor{keywordflow}{case} 1:
            printf(\textcolor{stringliteral}{"    responds to ping\(\backslash\)n"});
            \textcolor{keywordflow}{break};

        \textcolor{keywordflow}{default}:
            printf(\textcolor{stringliteral}{"    failed to ping, error %d [%s]\(\backslash\)n"}, errno,
                strerror(errno));
            \textcolor{keywordflow}{break};
        \}
    \}

    \hyperlink{bus_8h_a7a0ada5e8a9c567b1e7d18297e1cde42}{rpc\_bus\_free\_result}(nodes);
    \hyperlink{bus_8h_a5fa0c254755deec118b2e3dcb4ec9bdf}{rpc\_bus\_close}();
    \textcolor{keywordflow}{return} (0);
\}

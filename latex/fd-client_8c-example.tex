\textcolor{preprocessor}{#include <stdio.h>}
\textcolor{preprocessor}{#include <unistd.h>}
\textcolor{preprocessor}{#include <errno.h>}
\textcolor{preprocessor}{#include <string.h>}
\textcolor{preprocessor}{#include <\hyperlink{object_8h}{rpc/object.h}>}
\textcolor{preprocessor}{#include <\hyperlink{client_8h}{rpc/client.h}>}

\textcolor{keywordtype}{int}
main(\textcolor{keywordtype}{int} argc, \textcolor{keyword}{const} \textcolor{keywordtype}{char} *argv[])
\{
    \hyperlink{client_8h_a4f7230a0c2c07d27a07a88d407a06b70}{rpc\_client\_t} client;
    \hyperlink{connection_8h_a70838cb106c3464db299522c5fe2782d}{rpc\_connection\_t} conn;
    \hyperlink{object_8h_ab365f726b4975c0c8376b808d111d01b}{rpc\_object\_t} result;
    \textcolor{keywordtype}{int} fds[2];

    \textcolor{keywordflow}{if} (argc < 2) \{
        fprintf(stderr, \textcolor{stringliteral}{"Usage: fd-client <server socket URI>\(\backslash\)n"});
        \textcolor{keywordflow}{return} (1);
    \}

    client = \hyperlink{client_8h_aaf8c4f2be48ef8d79a29e004264f0bf6}{rpc\_client\_create}(argv[1], 0);
    \textcolor{keywordflow}{if} (client == NULL) \{
        fprintf(stderr, \textcolor{stringliteral}{"cannot connect: %s\(\backslash\)n"}, strerror(errno));
        \textcolor{keywordflow}{return} (1);
    \}

    \textcolor{keywordflow}{if} (pipe(fds) != 0) \{
        fprintf(stderr, \textcolor{stringliteral}{"cannot create pipe: %s\(\backslash\)n"}, strerror(errno));
        \textcolor{keywordflow}{return} (1);
    \}

    conn = \hyperlink{client_8h_a9e840cab717dbd16e7913f6c143c4804}{rpc\_client\_get\_connection}(client);
    result = \hyperlink{connection_8h_af5e8d63da05fa221ac8bb51ff22e7f3b}{rpc\_connection\_call\_sync}(conn, NULL, NULL, \textcolor{stringliteral}{"write\_to\_pipe"},
        \hyperlink{object_8h_aa2e6228045667b4e13b9897cfd2dc072}{rpc\_fd\_create}(fds[1]), NULL);
    printf(\textcolor{stringliteral}{"result = %s\(\backslash\)n"}, \hyperlink{object_8h_a04a698c1f5f736ef71bc81ccf91be52d}{rpc\_string\_get\_string\_ptr}(result));
    \hyperlink{object_8h_aac19be935f72b35eeb2851bc509eabb2}{rpc\_release}(result);

    close(fds[1]);

    \textcolor{keywordflow}{for} (;;) \{
        \textcolor{keywordtype}{char} buf[1024];
        ssize\_t ret;

        ret = read(fds[0], buf, \textcolor{keyword}{sizeof}(buf));
        \textcolor{keywordflow}{if} (ret <= 0)
            \textcolor{keywordflow}{break};

        printf(\textcolor{stringliteral}{"%.*s"}, (\textcolor{keywordtype}{int})ret, buf);
    \}

    \hyperlink{client_8h_ac4404bc6ff0f1d8be969acab48ba543b}{rpc\_client\_close}(client);
    \textcolor{keywordflow}{return} (0);
\}

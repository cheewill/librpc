\textcolor{preprocessor}{#include <stdio.h>}
\textcolor{preprocessor}{#include <unistd.h>}
\textcolor{preprocessor}{#include <errno.h>}
\textcolor{preprocessor}{#include <\hyperlink{object_8h}{rpc/object.h}>}
\textcolor{preprocessor}{#include <\hyperlink{service_8h}{rpc/service.h}>}
\textcolor{preprocessor}{#include <\hyperlink{server_8h}{rpc/server.h}>}

\textcolor{keyword}{static} \hyperlink{object_8h_ab365f726b4975c0c8376b808d111d01b}{rpc\_object\_t}
write\_to\_pipe(\textcolor{keywordtype}{void} *cookie, \hyperlink{object_8h_ab365f726b4975c0c8376b808d111d01b}{rpc\_object\_t} args)
\{
    \textcolor{keywordtype}{int} fd;

    \textcolor{keywordflow}{if} (\hyperlink{object_8h_a6ff01b9e6d71ba03e2dff76ede7366f6}{rpc\_object\_unpack}(args, \textcolor{stringliteral}{"[f]"}, &fd) < 1) \{
        \hyperlink{service_8h_a8199576424cc9470380ab43248129911}{rpc\_function\_error}(cookie, EINVAL, \textcolor{stringliteral}{"Invalid arguments passed"});
        \textcolor{keywordflow}{return} (NULL);
    \}

    printf(\textcolor{stringliteral}{"Received fd %d\(\backslash\)n"}, fd);

    dprintf(fd, \textcolor{stringliteral}{"Hello there\(\backslash\)n"});
    dprintf(fd, \textcolor{stringliteral}{"I am writing to the pipe\(\backslash\)n"});
    sleep(1);
    dprintf(fd, \textcolor{stringliteral}{"And sometimes sleeping\(\backslash\)n"});
    dprintf(fd, \textcolor{stringliteral}{"Bye.\(\backslash\)n"});
    close(fd);

    \textcolor{keywordflow}{return} (NULL);
\}

\textcolor{keywordtype}{int}
main(\textcolor{keywordtype}{int} argc, \textcolor{keyword}{const} \textcolor{keywordtype}{char} *argv[])
\{
    \hyperlink{service_8h_a55087f28cb63ff45b6c798a5dabaedd7}{rpc\_context\_t} ctx;
    \hyperlink{server_8h_a46fd27cbcf75a54103df9d51d90a7fca}{rpc\_server\_t} srv;

    (void)argc;
    (void)argv;

    ctx = \hyperlink{service_8h_aa53a346b1f60759fabc2f047e75fdb90}{rpc\_context\_create}();
    \hyperlink{service_8h_a1aac077f97529f7c26834c139fefc2f5}{rpc\_context\_register\_func}(ctx, NULL, \textcolor{stringliteral}{"write\_to\_pipe"}, NULL,
        write\_to\_pipe);
    srv = \hyperlink{server_8h_a2d524c3d5822d34b46f1d8552bf7ede0}{rpc\_server\_create}(\textcolor{stringliteral}{"unix:///tmp/server.sock"}, ctx);
    \hyperlink{server_8h_a52667e1f1ad29521541c60aae34dcf02}{rpc\_server\_resume}(srv);
    pause();
\}

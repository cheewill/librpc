#include <stdio.h>
#include <errno.h>
#include <rpc/object.h>
static rpc_object_t
fun_a(void)
{
return (rpc_error_create(ENOSYS, "It broke!", NULL));
}
static rpc_object_t
fun_b(void)
{
return (fun_a());
}
static rpc_object_t
fun_c(void)
{
return (fun_b());
}
int
main(int argc, const char *argv[])
{
rpc_object_t err;
(void)argc;
(void)argv;
err = fun_c();
printf("%s\n", rpc_copy_description(err));
return (0);
}
